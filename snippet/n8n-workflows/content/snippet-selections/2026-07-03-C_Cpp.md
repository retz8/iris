std::string unescape(const std::string_view data)
{
    std::string result{};
    result.reserve(data.size());

    bool xor_next = false;

    for (auto value : data)
    {
        if (xor_next)
        {
            value ^= 0x20;
            xor_next = false;
        }
        else if (value == '}')
        {
            xor_next = true;
            continue;
        }

        result.push_back(value);
    }

    rt_assert(!xor_next);

    return result;
}
