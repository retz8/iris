template<bool VALUE>
Optional<size_t> find_first() const
{
    size_t byte_count = size_in_bytes();
    size_t i = 0;
    u8 byte = VALUE ? 0x00 : 0xff;
    while (i < byte_count && m_data[i] == byte)
        i++;
    if (i == byte_count)
        return {};
    byte = m_data[i];
    if constexpr (!VALUE)
        byte = ~byte;
    VERIFY(byte != 0);
    return i * 8 + bit_scan_forward(byte) - 1;
}
