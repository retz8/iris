void nixlSecDescList::addDesc(
        const nixlSectionDesc &desc) {
    auto &vec = this->descs;
    auto itr = std::upper_bound(
        vec.begin(), vec.end(), desc);
    if (itr == vec.end())
        vec.push_back(desc);
    else
        vec.insert(itr, desc);
}
